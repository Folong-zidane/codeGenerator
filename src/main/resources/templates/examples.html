Commande de génération
# Spring Boot
python crud_generator_cli.py generate \
  --uml ecommerce.mermaid \
  --output ./ecommerce-api \
  --framework spring-boot \
  --package com.ecommerce \
  --type complete-project

    }

        +UUID id
        +UUID postId
        +String authorName
        +String content
        +Date createdAt
        +moderate()
    }
    
    Author "1" --> "*" Post
    Post "1" --> "*" Comment
Commande
python crud_generator_cli.py generate \
  --uml blog.mermaid \
  --output ./blog-django \
  --framework django \
  --package blog \
  --type complete-project
Lancement
cd blog-django
pip install -r requirements.txt
python manage.py migrate
python manage.py runserver
# API sur http://localhost:8000

  --uml mon-diagram.mermaid
Génération fichiers uniquement
# Juste les classes CRUD

  --uml diagram.mermaid \
  --output ./crud-only \
  --type files-only
